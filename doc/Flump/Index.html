def self.body
  <<~HTML
    <!doctype html>
    <html lang="en">

    <head>
      <meta charset="utf-8">
      <title>#{TITLE}</title>
        <meta name="description" content="">
        <meta name="viewport" content="width=device-width, initial-scale=1">
        <style>
          * {
            background-color: black;
            border: 0;
            color: green;
            font-family: "Courier New";
            font-size: 16pt;
            margin: 0px;
            padding: 0;
          }

          body {
            display: flex;
            flex-flow: column;
            height: 100vh;
          }

          .chat {
            overflow-y: scroll;
            flex: 1 1 auto;
          }

          .chat-input {
            width: 100vw;
          }
        </style>
        <script>
          window.onload = () => {
            const webSocket = new WebSocket("ws://#{HOST}:#{PORT}");

            webSocket.onopen = () => console.log('connected');
            webSocket.onclose = () => console.error('disconnected');
            webSocket.onerror = error => console.error('failed to connect', error);

            webSocket.onmessage = event => {
              console.log('received', event.data);
              const messageElement = document.createElement('div');
              messageElement.className = 'message';
              messageElement.innerText = event.data;
              document.querySelector('.chat').appendChild(messageElement);
              messageElement.scrollIntoView();
            };

            const onSubmit = event => {
              event.preventDefault();
              const message = document.querySelector('.chat-input').value;
              webSocket.send(message);
              document.querySelector('.chat-input').value = '';
            };

            document.querySelector('form').addEventListener('submit', onSubmit);
          };
        </script>
    </head>

    <body>
      <div class="chat"></div>
      <form>
        <input class="chat-input" autofocus required>
      </form>
    </body>

    </html>
  HTML
end
