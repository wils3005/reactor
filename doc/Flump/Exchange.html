def initialize(str)
  @request = str
  @headers, @body = @request.split(CRLF * 2)
  @headers = @headers.split(CRLF)
  @request_line = @headers.shift.split
  @method, @path, @version = @request_line
  @path, @query = @path.split('?')
  @path = @path.split('/')
  @query = @query.split('&').map { _1.split('=') }.to_h if @query
  @headers = @headers.map { _1.split(': ') }.to_h
  @status_code, @reason_phrase = _status_code_with_reason_phrase
  @response = _response
end
